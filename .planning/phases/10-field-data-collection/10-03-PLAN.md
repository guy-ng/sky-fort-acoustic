<!-- API contract from Plan 02 -->

REST endpoints:
- POST /api/recordings/start -> { id: string, status: "recording" }
- POST /api/recordings/stop -> { id: string, duration_s: number, status: "stopped" }
- GET /api/recordings -> Array<{ id, label, labeled, duration_s, recorded_at, sub_label?, ... }>
- POST /api/recordings/{id}/label -> body: { label, sub_label?, distance_m?, altitude_m?, conditions?, notes? }
- PATCH /api/recordings/{id} -> body: { sub_label?, distance_m?, altitude_m?, conditions?, notes? }
- DELETE /api/recordings/{id} -> { id, deleted: true }

WebSocket /ws/recording -> { status: "idle"|"recording", elapsed_s: number, remaining_s: number, level_db: number }

    ```typescript
    // Types
    export interface Recording {
      id: string
      label: string
      labeled: boolean
      duration_s: number
      recorded_at: string
      sub_label?: string
      distance_m?: number
      altitude_m?: number
      conditions?: string
      notes?: string
      filename?: string
    }

    // Hooks
    export function useRecordingsList() {
      return useQuery({ queryKey: ['recordings'], queryFn: () => fetch('/api/recordings').then(r => r.json()) })
    }

    export function useStartRecording() {
      const qc = useQueryClient()
      return useMutation({
        mutationFn: () => fetch('/api/recordings/start', { method: 'POST' }).then(r => r.json()),
        onSuccess: () => qc.invalidateQueries({ queryKey: ['recordings'] })
      })
    }

    export function useStopRecording() {
      const qc = useQueryClient()
      return useMutation({
        mutationFn: () => fetch('/api/recordings/stop', { method: 'POST' }).then(r => r.json()),
        onSuccess: () => qc.invalidateQueries({ queryKey: ['recordings'] })
      })
    }

    export function useLabelRecording() {
      const qc = useQueryClient()
      return useMutation({
        mutationFn: ({ id, body }: { id: string, body: { label: string, sub_label?: string, distance_m?: number, altitude_m?: number, conditions?: string, notes?: string } }) =>
          fetch(`/api/recordings/${id}/label`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(body) }).then(r => r.json()),
        onSuccess: () => qc.invalidateQueries({ queryKey: ['recordings'] })
      })
    }

    export function useUpdateRecording() { ... }  // PATCH /api/recordings/{id}
    export function useDeleteRecording() { ... }  // DELETE /api/recordings/{id}
    ```

    **AudioLevelMeter.tsx** -- Horizontal level bar (per UI-SPEC):
    ```tsx
    interface AudioLevelMeterProps { level_db: number }
    export function AudioLevelMeter({ level_db }: AudioLevelMeterProps) {
      // Map -60dB..0dB to 0%..100% width
      const pct = Math.max(0, Math.min(100, ((level_db + 60) / 60) * 100))
      // Color: hud-success (low) -> hud-warning (mid) -> hud-danger (high)
      const color = level_db > -6 ? 'bg-hud-danger' : level_db > -20 ? 'bg-hud-warning' : 'bg-hud-success'
      return (
        <div className="w-full h-1.5 bg-hud-border rounded-full overflow-hidden">
          <div className={`h-full ${color} transition-all duration-100`} style={{ width: `${pct}%` }} />
        </div>
      )
    }
    ```
