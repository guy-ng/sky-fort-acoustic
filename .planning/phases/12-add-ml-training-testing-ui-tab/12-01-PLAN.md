```python
class TrainingStartRequest(BaseModel):
    learning_rate: float | None = None
    batch_size: int | None = None
    max_epochs: int | None = None
    patience: int | None = None
    augmentation_enabled: bool | None = None
    data_root: str | None = None

class ConfusionMatrixResponse(BaseModel):
    tp: int
    fp: int
    tn: int
    fn: int

class TrainingProgressResponse(BaseModel):
    status: str  # idle/running/completed/cancelled/failed
    epoch: int
    total_epochs: int
    train_loss: float
    val_loss: float
    val_acc: float
    best_val_loss: float
    best_epoch: int = 0
    confusion_matrix: ConfusionMatrixResponse
    error: str | None = None

class EvalRunRequest(BaseModel):
    model_path: str | None = None
    data_dir: str | None = None
    ensemble_config_path: str | None = None

class EvalSummaryResponse(BaseModel):
    total: int
    correct: int
    incorrect: int
    accuracy: float
    precision: float
    recall: float
    f1: float
    confusion_matrix: ConfusionMatrixResponse

class DistributionStatsResponse(BaseModel):
    p25: float; p50: float; p75: float; p95: float

class ClassDistributionResponse(BaseModel):
    p_agg: DistributionStatsResponse
    p_max: DistributionStatsResponse
    p_mean: DistributionStatsResponse

class FileResultResponse(BaseModel):
    filename: str
    true_label: str
    predicted_label: str
    p_agg: float
    correct: bool

class PerModelResultResponse(BaseModel):
    model_type: str
    model_path: str
    weight: float
    accuracy: float
    precision: float
    recall: float
    f1: float

class EvalResultResponse(BaseModel):
    summary: EvalSummaryResponse
    distribution: dict[str, ClassDistributionResponse]
    per_file: list[FileResultResponse]
    model_path: str
    data_dir: str
    message: str
    per_model_results: list[PerModelResultResponse] | None = None

class ModelInfo(BaseModel):
    filename: str
    path: str
    size_bytes: int
    modified: str  # ISO 8601

class ModelListResponse(BaseModel):
    models: list[ModelInfo]
```

- Completed: same as running but `"status": "completed"`
- Failed: `{ "status": "failed", "error": "Out of memory" }`

- `useTrainingProgress()`: useQuery with queryKey `['training-progress']`, fetches `GET /api/training/progress`, returns `TrainingProgressResponse`. Set `refetchInterval: 5000` as polling fallback (primary updates come via WebSocket, this is for initial load and fallback).
- Import types from `../../utils/types`.

**web/src/hooks/useTrainingSocket.ts:**
- Copy the structure of useRecordingSocket.ts exactly (wsRef, reconnectTimer, reconnectDelay, connect callback, useEffect with cleanup).
- Connect to `/ws/training` (same protocol detection: `ws:` or `wss:` based on `window.location.protocol`).
- State: `trainingState` of type `TrainingWsMessage` initialized to `{ status: 'idle' }`.
- Additionally maintain a `lossHistory` ref of type `LossDataPoint[]` that accumulates `{ epoch, train_loss, val_loss }` on each 'running' message. Clear the array when status transitions to 'running' and epoch is 1 (new run starting).
- Return object: `{ state: TrainingWsMessage, lossHistory: LossDataPoint[] }`.
- Use `useState` for lossHistory (not just ref) so re-renders trigger when new data arrives.
