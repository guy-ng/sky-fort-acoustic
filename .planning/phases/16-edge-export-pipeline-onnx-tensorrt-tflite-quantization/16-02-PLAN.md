    ```python
    """TensorRT FP16/INT8 engine conversion (optional -- requires NVIDIA GPU + tensorrt)."""
    from __future__ import annotations
    import logging
    import os

    logger = logging.getLogger(__name__)

    def convert_to_tensorrt(
        onnx_path: str,
        output_path: str,
        precision: str = "fp16",  # "fp16" or "int8"
        max_batch_size: int = 1,
        calibration_data: "np.ndarray | None" = None,
    ) -> dict:
        """Convert ONNX model to TensorRT engine.

        Returns {"status": "success", "path": output_path} on success,
        or {"status": "skipped", "reason": "..."} if tensorrt not available.
        """
        try:
            import tensorrt as trt
        except ImportError:
            return {"status": "skipped", "reason": "tensorrt not installed"}

        TRT_LOGGER = trt.Logger(trt.Logger.WARNING)
        builder = trt.Builder(TRT_LOGGER)
        network = builder.create_network(1 << int(trt.NetworkDefinitionCreationFlag.EXPLICIT_BATCH))
        parser = trt.OnnxParser(network, TRT_LOGGER)

        with open(onnx_path, "rb") as f:
            if not parser.parse(f.read()):
                errors = [parser.get_error(i) for i in range(parser.num_errors)]
                return {"status": "error", "reason": f"ONNX parse failed: {errors}"}

        config = builder.create_builder_config()
        config.set_memory_pool_limit(trt.MemoryPoolType.WORKSPACE, 1 << 30)  # 1GB

        if precision == "fp16":
            config.set_flag(trt.BuilderFlag.FP16)
        elif precision == "int8":
            config.set_flag(trt.BuilderFlag.INT8)
            # INT8 calibration would be set here with calibration_data

        engine_bytes = builder.build_serialized_network(network, config)
        if engine_bytes is None:
            return {"status": "error", "reason": "TensorRT engine build failed"}

        with open(output_path, "wb") as f:
            f.write(engine_bytes)

        logger.info("TensorRT %s engine saved to %s", precision, output_path)
        return {"status": "success", "path": output_path, "precision": precision}
    ```

    ```python
    """TFLite INT8 conversion via onnx2tf (optional -- requires onnx2tf pip package)."""
    from __future__ import annotations
    import logging
    import os

    logger = logging.getLogger(__name__)

    def convert_to_tflite(
        onnx_path: str,
        output_dir: str,
        quantize_int8: bool = True,
        calibration_data_path: str | None = None,
    ) -> dict:
        """Convert ONNX model to TFLite with optional INT8 quantization.

        Returns {"status": "success", "path": output_dir} on success,
        or {"status": "skipped", "reason": "..."} if onnx2tf not available.
        """
        try:
            import onnx2tf
        except ImportError:
            return {"status": "skipped", "reason": "onnx2tf not installed"}

        os.makedirs(output_dir, exist_ok=True)

        kwargs = {
            "input_onnx_file_path": onnx_path,
            "output_folder_path": output_dir,
            "copy_onnx_input_output_names_to_tflite": True,
            "non_verbose": True,
        }

        if quantize_int8:
            kwargs["output_integer_quantized_tflite"] = True
            if calibration_data_path and os.path.exists(calibration_data_path):
                kwargs["quant_calib_input_op_name_np_data_path"] = calibration_data_path

        onnx2tf.convert(**kwargs)

        logger.info("TFLite model saved to %s", output_dir)
        return {"status": "success", "path": output_dir, "quantized": quantize_int8}
    ```
