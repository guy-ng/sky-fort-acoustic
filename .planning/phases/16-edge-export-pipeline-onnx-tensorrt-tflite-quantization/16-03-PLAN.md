    Create `src/acoustic/api/export_routes.py`:
    ```python
    """REST API for model export (ONNX, TensorRT, TFLite)."""
    from __future__ import annotations

    import os
    import uuid
    import asyncio
    import logging
    from typing import Dict

    from fastapi import APIRouter, HTTPException

    from acoustic.api.models import ExportRequest, ExportJobStatus, ExportResponse
    from acoustic.export.exporter import ExportPipeline

    logger = logging.getLogger(__name__)
    router = APIRouter(prefix="/api/export", tags=["export"])

    # In-memory job store (sufficient for single-instance service)
    _jobs: Dict[str, ExportJobStatus] = {}
    _pipeline = ExportPipeline()

    @router.post("", response_model=ExportResponse, status_code=202)
    async def start_export(request: ExportRequest) -> ExportResponse:
        # Validate model_type
        if request.model_type not in ExportPipeline.SUPPORTED_MODELS:
            raise HTTPException(400, f"Invalid model_type. Must be one of: {ExportPipeline.SUPPORTED_MODELS}")
        # Validate formats
        invalid_formats = set(request.formats) - ExportPipeline.SUPPORTED_FORMATS
        if invalid_formats:
            raise HTTPException(400, f"Invalid formats: {invalid_formats}")
        # Validate checkpoint exists
        if not os.path.isfile(request.checkpoint_path):
            raise HTTPException(404, f"Checkpoint not found: {request.checkpoint_path}")

        job_id = str(uuid.uuid4())
        output_dir = request.output_dir or os.path.join("models", "export", job_id)

        job = ExportJobStatus(
            job_id=job_id, status="pending",
            model_type=request.model_type, formats=request.formats,
        )
        _jobs[job_id] = job

        # Run export in background thread (CPU-bound work)
        asyncio.get_event_loop().run_in_executor(
            None, _run_export, job_id, request.checkpoint_path, output_dir,
            request.model_type, request.formats,
        )
        return ExportResponse(job_id=job_id, status="pending")

    def _run_export(job_id: str, checkpoint_path: str, output_dir: str, model_type: str, formats: list[str]) -> None:
        job = _jobs[job_id]
        job.status = "running"
        try:
            results = _pipeline.export(checkpoint_path, output_dir, model_type, formats)
            job.results = results
            job.status = "completed"
        except Exception as e:
            logger.exception("Export job %s failed", job_id)
            job.error = str(e)
            job.status = "failed"

    @router.get("/{job_id}", response_model=ExportJobStatus)
    async def get_export_status(job_id: str) -> ExportJobStatus:
        if job_id not in _jobs:
            raise HTTPException(404, f"Export job not found: {job_id}")
        return _jobs[job_id]
    ```
