<!-- API contract from Plan 01 that this frontend consumes -->

WebSocket /ws/heatmap protocol:
1. Server sends JSON handshake: {"type": "handshake", "width": 181, "height": 91, "az_min": -90, "az_max": 90, "el_min": -45, "el_max": 45}
2. Server sends binary frames: raw float32 bytes, width*height values, row-major (elevation rows, azimuth columns)
3. Client must set ws.binaryType = "arraybuffer"

WebSocket /ws/targets protocol:
1. Server sends JSON array of targets: [{"id": "uuid", "class_label": "unknown", "speed_mps": null, "az_deg": 30.0, "el_deg": 10.0, "confidence": 0.7}]
2. Updates at ~2 Hz

REST GET /api/map response:
```json
{
  "az_min": -90.0, "az_max": 90.0,
  "el_min": -45.0, "el_max": 45.0,
  "az_resolution": 1.0, "el_resolution": 1.0,
  "width": 181, "height": 91,
  "data": [[...], ...],
  "peak": {"az_deg": 30.0, "el_deg": 10.0, "power": 0.85}
}
```

REST GET /api/targets response:
```json
[{"id": "uuid", "class_label": "unknown", "speed_mps": null, "az_deg": 30.0, "el_deg": 10.0, "confidence": 0.7}]
```

REST GET /health response:
```json
{"status": "ok", "device_detected": false, "pipeline_running": true, "overflow_count": 0, "last_frame_time": 1234567.89}
```

```typescript
export function jetColormap(t: number): [number, number, number] {
  t = Math.max(0, Math.min(1, t))
  let r: number, g: number, b: number

  if (t < 0.125) {
    r = 0; g = 0; b = 0.5 + t * 4
  } else if (t < 0.375) {
    r = 0; g = (t - 0.125) * 4; b = 1
  } else if (t < 0.625) {
    r = (t - 0.375) * 4; g = 1; b = 1 - (t - 0.375) * 4
  } else if (t < 0.875) {
    r = 1; g = 1 - (t - 0.625) * 4; b = 0
  } else {
    r = 1 - (t - 0.875) * 4; g = 0; b = 0
  }

  return [Math.round(r * 255), Math.round(g * 255), Math.round(b * 255)]
}
```
