Create `src/acoustic/api/websocket.py` with an APIRouter:

1. `WS /ws/heatmap` (per API-03, D-06, D-07):
   - Accept connection
   - Send JSON handshake message with grid dimensions (HeatmapHandshake model) using `send_json()`
   - Enter async loop: poll `app.state.pipeline.latest_map` with `asyncio.sleep(0.05)` (20 Hz poll)
   - Track last map via `id(current_map)` to avoid resending same frame
   - When new map available, send `current_map.astype(np.float32).tobytes()` via `send_bytes()`
   - The map must be transposed to row-major order before tobytes: `current_map.T.astype(np.float32).tobytes()`
   - Wrap in try/except for WebSocketDisconnect and RuntimeError (per Pitfall 1)

2. `WS /ws/targets` (per D-07):
   - Accept connection
   - Enter async loop: poll `app.state.pipeline.latest_peak` with `asyncio.sleep(0.5)` (2 Hz -- targets change slowly)
   - When peak exists, send JSON array with one placeholder target via `send_json()`
   - When no peak, send empty array `[]`
   - Wrap in try/except for WebSocketDisconnect

```python
import struct
import numpy as np
from starlette.testclient import TestClient

class TestHeatmapWebSocket:
    def test_heatmap_ws_sends_handshake(self, running_app):
        # Connect to /ws/heatmap, first message is JSON with type="handshake"
        client = TestClient(running_app)
        with client.websocket_connect("/ws/heatmap") as ws:
            data = ws.receive_json()
            assert data["type"] == "handshake"
            assert "width" in data
            assert "height" in data

    def test_heatmap_ws_sends_binary(self, running_app):
        # After handshake, receive binary float32 frame
        client = TestClient(running_app)
        with client.websocket_connect("/ws/heatmap") as ws:
            handshake = ws.receive_json()
            width = handshake["width"]
            height = handshake["height"]
            data = ws.receive_bytes(timeout=5)
            values = np.frombuffer(data, dtype=np.float32)
            assert len(values) == width * height

class TestTargetsWebSocket:
    def test_targets_ws_sends_json(self, running_app):
        # Connect to /ws/targets, receive JSON array
        client = TestClient(running_app)
        with client.websocket_connect("/ws/targets") as ws:
            data = ws.receive_json()
            assert isinstance(data, list)
```

Note: Starlette TestClient WebSocket is synchronous (not async). Use `TestClient` not `AsyncClient` for WebSocket tests.
