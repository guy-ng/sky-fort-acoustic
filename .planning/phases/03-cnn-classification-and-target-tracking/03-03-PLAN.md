    **1. Add CNNWorker class to `src/acoustic/pipeline.py`:**

    Add a `CNNWorker` class (port of POC's CNNWorker pattern per Research Pattern 1):
    ```python
    import queue
    from dataclasses import dataclass
    from acoustic.classification.inference import OnnxDroneClassifier
    from acoustic.classification.state_machine import DetectionStateMachine, DetectionState
    from acoustic.tracking.tracker import TargetTracker, TrackedTarget
    from acoustic.tracking.publisher import ZmqPublisher
    from acoustic.tracking.schema import TargetEvent

    @dataclass
    class ClassificationResult:
        drone_probability: float
        timestamp: float
        az_deg: float
        el_deg: float

    class CNNWorker:
        def __init__(
            self,
            classifier: OnnxDroneClassifier,
            state_machine: DetectionStateMachine,
            tracker: TargetTracker,
            publisher: ZmqPublisher | None,
            sample_rate: int = 48000,
        ):
            self._classifier = classifier
            self._state_machine = state_machine
            self._tracker = tracker
            self._publisher = publisher
            self._sample_rate = sample_rate
            self._queue: queue.Queue = queue.Queue(maxsize=1)
            self._latest: ClassificationResult | None = None
            self._lock = threading.Lock()
            self._stop = threading.Event()
            self._thread = threading.Thread(target=self._loop, daemon=True)

        @property
        def latest(self) -> ClassificationResult | None:
            with self._lock:
                return self._latest

        def push(self, mono_audio: np.ndarray, az_deg: float, el_deg: float) -> None:
            """Non-blocking push. Drops if queue full (latest-only semantics)."""
            try:
                self._queue.put_nowait((mono_audio, az_deg, el_deg))
            except queue.Full:
                pass

        def _loop(self) -> None:
            while not self._stop.is_set():
                try:
                    mono_audio, az, el = self._queue.get(timeout=0.1)
                except queue.Empty:
                    continue
                try:
                    prob = self._classifier.predict_from_audio(mono_audio, self._sample_rate)
                    result = ClassificationResult(
                        drone_probability=prob,
                        timestamp=time.monotonic(),
                        az_deg=az, el_deg=el,
                    )
                    with self._lock:
                        self._latest = result

                    # Feed through state machine
                    state = self._state_machine.update(prob)

                    # Update tracker based on state
                    if state == DetectionState.CONFIRMED:
                        class_label = "drone" if prob >= 0.5 else "background"
                        event_type, target = self._tracker.update_target(
                            class_label=class_label,
                            confidence=prob,
                            az_deg=az,
                            el_deg=el,
                        )
                        # Publish ZMQ event
                        if self._publisher is not None:
                            self._publisher.publish(TargetEvent(
                                event=event_type,
                                target_id=target.target_id,
                                class_label=target.class_label,
                                confidence=target.confidence,
                                az_deg=az, el_deg=el,
                                speed_mps=None,
                                timestamp=time.time(),
                            ))

                    # Expire lost targets
                    lost = self._tracker.expire_lost_targets()
                    if self._publisher is not None:
                        for t in lost:
                            self._publisher.publish(TargetEvent(
                                event="lost",
                                target_id=t.target_id,
                                class_label=t.class_label,
                                confidence=0.0,
                                az_deg=t.az_deg, el_deg=t.el_deg,
                                speed_mps=None,
                                timestamp=time.time(),
                            ))
                except Exception:
                    logger.exception("Error in CNN worker inference")

        def start(self) -> None:
            self._thread.start()

        def stop(self) -> None:
            self._stop.set()
            self._thread.join(timeout=5.0)
    ```
