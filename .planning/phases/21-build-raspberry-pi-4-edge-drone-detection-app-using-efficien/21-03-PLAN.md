3. Wrap the model in a `LogitsOnly` adapter that discards the `(logits, features)` tuple and returns only logits:
```python
import torch.nn as nn
class LogitsOnly(nn.Module):
    def __init__(self, mn):
        super().__init__()
        self.mn = mn
    def forward(self, mel):
        out = self.mn(mel)
        if isinstance(out, tuple):
            return out[0]
        return out
```

4. Generate 20 held-out calibration samples: torch.randn((1, 1, 128, 100), dtype=torch.float32) * 0.5 — these simulate mel spectrograms in the trained distribution range. Also generate a `reference_batch` of 20 stacked samples for the sanity gate. Seed with `torch.manual_seed(42)` for determinism.

5. Export FP32 ONNX via `torch.onnx.export`:
```python
dummy = torch.randn(1, 1, 128, 100)  # (batch, channel, n_mels, n_frames)
torch.onnx.export(
    logits_only_model,
    (dummy,),
    str(output_dir / "efficientat_mn10_v6_fp32.onnx"),
    input_names=["mel"],
    output_names=["logits"],
    opset_version=17,
    dynamo=False,
    dynamic_axes={"mel": {0: "batch"}, "logits": {0: "batch"}},
)
import onnx
onnx.checker.check_model(str(output_dir / "efficientat_mn10_v6_fp32.onnx"))
```

```python
"""D-06, D-07, D-08: ONNX conversion sanity gate (re-verification from committed artifacts)."""
import hashlib
import numpy as np
import pytest
from pathlib import Path

REPO_ROOT = Path(__file__).resolve().parents[3]
FP32_ONNX = REPO_ROOT / "models/efficientat_mn10_v6_fp32.onnx"
INT8_ONNX = REPO_ROOT / "models/efficientat_mn10_v6_int8.onnx"
CHECKSUMS = REPO_ROOT / "models/efficientat_mn10_v6_onnx.sha256"
CHECKPOINT = REPO_ROOT / "models/efficientat_mn10_v6.pt"

N_SAMPLES = 50
RNG_SEED = 1337


def _make_ref_batch():
    rng = np.random.default_rng(RNG_SEED)
    return rng.standard_normal((N_SAMPLES, 1, 128, 100)).astype(np.float32) * 0.5


def _torch_logits(batch):
    torch = pytest.importorskip("torch")
    import sys
    sys.path.insert(0, str(REPO_ROOT))
    from src.acoustic.classification.efficientat.model import get_model
    from src.acoustic.classification.efficientat import config as mn_config
    # Use the same construction as the conversion script — keep in sync!
    state = torch.load(str(CHECKPOINT), map_location="cpu", weights_only=True)
    # ... construct MN via model.py helper, load state_dict, eval, forward batch
    # (mirror the exact construction from scripts/convert_efficientat_to_onnx.py Task 1)
    model = get_model(...)  # fill per model.py signature; match conversion script
    model.load_state_dict(state if isinstance(state, dict) else state["state_dict"])
    model.eval()
    with torch.no_grad():
        out = model(torch.from_numpy(batch))
        if isinstance(out, tuple):
            out = out[0]
    return out.numpy()


def _onnx_logits(onnx_path, batch):
    ort = pytest.importorskip("onnxruntime")
    sess = ort.InferenceSession(str(onnx_path), providers=["CPUExecutionProvider"])
    input_name = sess.get_inputs()[0].name
    return sess.run(None, {input_name: batch})[0]


@pytest.fixture(scope="module")
def ref_batch():
    return _make_ref_batch()


@pytest.fixture(scope="module")
def torch_reference(ref_batch):
    return _torch_logits(ref_batch)


def test_fp32_onnx_top1_agreement_ge_99pct(ref_batch, torch_reference):
    assert FP32_ONNX.exists(), f"missing {FP32_ONNX} — run scripts/convert_efficientat_to_onnx.py first"
    onnx_out = _onnx_logits(FP32_ONNX, ref_batch)
    torch_top1 = np.argmax(torch_reference, axis=-1)
    onnx_top1 = np.argmax(onnx_out, axis=-1)
    agreement = float(np.mean(torch_top1 == onnx_top1))
    assert agreement >= 0.99, f"FP32 top-1 agreement {agreement:.4f} < 0.99"


def test_int8_onnx_top1_agreement_ge_95pct(ref_batch, torch_reference):
    assert INT8_ONNX.exists(), f"missing {INT8_ONNX}"
    onnx_out = _onnx_logits(INT8_ONNX, ref_batch)
    torch_top1 = np.argmax(torch_reference, axis=-1)
    onnx_top1 = np.argmax(onnx_out, axis=-1)
    agreement = float(np.mean(torch_top1 == onnx_top1))
    assert agreement >= 0.95, f"int8 top-1 agreement {agreement:.4f} < 0.95 (research note: Conv layers are NOT quantized by quantize_dynamic)"


def test_mean_logit_delta_below_threshold(ref_batch, torch_reference):
    onnx_out = _onnx_logits(FP32_ONNX, ref_batch)
    delta = float(np.mean(np.abs(onnx_out - torch_reference)))
    assert delta < 0.05, f"mean logit delta {delta:.4f} >= 0.05"


def test_checksum_file_valid():
    assert CHECKSUMS.exists()
    expected = {}
    for line in CHECKSUMS.read_text().splitlines():
        parts = line.strip().split(maxsplit=1)
        if len(parts) == 2:
            expected[parts[1].strip()] = parts[0].strip()
    for onnx_file in [FP32_ONNX, INT8_ONNX]:
        actual = hashlib.sha256(onnx_file.read_bytes()).hexdigest()
        name = onnx_file.name
        assert name in expected, f"no checksum recorded for {name}"
        assert actual == expected[name], f"sha256 mismatch for {name}: expected {expected[name]}, got {actual}"
```
