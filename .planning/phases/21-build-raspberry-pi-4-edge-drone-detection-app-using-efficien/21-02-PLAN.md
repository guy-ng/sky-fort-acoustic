class AugmentMelSTFT:
    sr = 32000, n_mels = 128, win_length = 800, hopsize = 320, n_fft = 1024
    preemphasis_coefficient = [-0.97, 1]  # applied via conv1d with kernel (1,1,2)
    window = torch.hann_window(800, periodic=False)
    mel_basis = torch.load("mel_banks_128_1024_32k.pt")  # shape (128, 513) — n_fft//2+1 after F.pad with 0

    forward(x):
      x = conv1d(x.unsqueeze(1), preemphasis).squeeze(1)   # preemphasis filter
      x = torch.stft(x, n_fft=1024, hop=320, win=800, center=True,
                     window=hann_window_periodic_false, return_complex=False)  # (B, F, T, 2)
      x = (x**2).sum(-1)                                    # power magnitude  (B, F, T)
      mel = mel_basis @ x                                   # (B, 128, T)
      mel = log(mel + 0.00001)
      mel = (mel + 4.5) / 5.0
      return mel  # shape (B, 128, T)

Inference mode (self.training == False): freqm / timem are NO-OPs — the numpy port must skip them entirely.

The torch hann_window(periodic=False) of length N equals numpy:
    np.hanning(N)  is symmetric (equivalent to periodic=False)
BUT note np.hanning returns (N,) starting and ending near 0; torch.hann_window(periodic=False) = 0.5 - 0.5*cos(2*pi*n/(N-1)) for n=0..N-1, which matches np.hanning exactly.

torch.stft with center=True uses reflect padding of n_fft//2 on both sides (this is PyTorch's default for stft).

```python
"""Pure-numpy port of EfficientAT AugmentMelSTFT (inference mode only).

Parity-locked against src/acoustic/classification/efficientat/preprocess.py via
apps/rpi-edge/tests/test_preprocess_parity.py (atol=1e-5). Do NOT edit without
updating the parity test expectations.

Intentionally no torch / torchaudio imports — D-28 forbids PyTorch on the Pi.
"""
from __future__ import annotations
from pathlib import Path
import numpy as np

_MEL_BANKS_PATH = Path(__file__).parent / "mel_banks_128_1024_32k.npy"

SR = 32000
N_MELS = 128
WIN_LENGTH = 800
HOP_SIZE = 320
N_FFT = 1024
PREEMPH = np.array([-0.97, 1.0], dtype=np.float32)  # [-0.97, 1] per torch conv1d kernel


def _hann_window_symmetric(n: int) -> np.ndarray:
    # Matches torch.hann_window(n, periodic=False) exactly: 0.5 - 0.5*cos(2*pi*k/(n-1))
    k = np.arange(n, dtype=np.float32)
    return (0.5 - 0.5 * np.cos(2.0 * np.pi * k / (n - 1))).astype(np.float32)


class NumpyMelSTFT:
    def __init__(self) -> None:
        self.window = _hann_window_symmetric(WIN_LENGTH)
        mel_basis = np.load(_MEL_BANKS_PATH)
        assert mel_basis.shape == (N_MELS, N_FFT // 2 + 1), (
            f"mel_basis shape {mel_basis.shape} != ({N_MELS}, {N_FFT // 2 + 1})"
        )
        self.mel_basis = mel_basis.astype(np.float32)

    def _preemphasis(self, x: np.ndarray) -> np.ndarray:
        # torch conv1d with kernel [-0.97, 1], no padding: out[t] = -0.97*x[t] + 1*x[t+1]
        # Output length = len(x) - 1
        return (-0.97 * x[:-1] + x[1:]).astype(np.float32)

    def _stft(self, x: np.ndarray) -> np.ndarray:
        # torch.stft with center=True reflect-pads n_fft//2 on both sides
        pad = N_FFT // 2
        xp = np.pad(x, (pad, pad), mode="reflect")
        # Frame it
        num_frames = 1 + (len(xp) - N_FFT) // HOP_SIZE
        frames = np.zeros((num_frames, N_FFT), dtype=np.float32)
        for i in range(num_frames):
            start = i * HOP_SIZE
            seg = xp[start : start + N_FFT]
            # Apply window (length WIN_LENGTH, centered in N_FFT? No — torch applies window of length win_length
            # directly, then zero-pads to n_fft)
            windowed = np.zeros(N_FFT, dtype=np.float32)
            # torch.stft pads win_length to n_fft by centering the window (win_length <= n_fft)
            win_pad_left = (N_FFT - WIN_LENGTH) // 2
            windowed[win_pad_left : win_pad_left + WIN_LENGTH] = (
                seg[win_pad_left : win_pad_left + WIN_LENGTH] * self.window
            )
            frames[i] = windowed
        # rfft
        spec = np.fft.rfft(frames, n=N_FFT, axis=-1)  # (T, F)
        # Power magnitude: re^2 + im^2
        power = (spec.real ** 2 + spec.imag ** 2).astype(np.float32)
        return power.T  # (F, T)

    def forward(self, wave: np.ndarray) -> np.ndarray:
        """wave: 1-D float32 array at 32 kHz. Returns (128, T) float32 mel."""
        if wave.ndim != 1:
            raise ValueError(f"expected 1-D wave, got shape {wave.shape}")
        wave = wave.astype(np.float32)
        x = self._preemphasis(wave)
        power = self._stft(x)                         # (F, T)
        mel = self.mel_basis @ power                  # (128, T)
        mel = np.log(mel + 1e-5)
        mel = (mel + 4.5) / 5.0
        return mel.astype(np.float32)
```

IMPORTANT — window placement inside the STFT frame is subtle. torch.stft with `win_length < n_fft` centers the window inside the n_fft-length frame (win_pad_left = (n_fft - win_length) // 2) and leaves the rest zero. The code above matches that. If the parity test fails, revisit this exact block.

import numpy as np
import pytest
import soundfile as sf
from pathlib import Path
from scipy.signal import resample_poly

from skyfort_edge.preprocess import NumpyMelSTFT

FIXTURES = Path(__file__).parent / "fixtures"

@pytest.mark.parametrize("fixture_name", ["golden_drone_1s_48k.wav", "golden_silence_1s_48k.wav"])

    # Torch reference (host-only import, skipped on Pi)

import soundfile as sf
