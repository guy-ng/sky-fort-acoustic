[Unit]
Description=Sky Fort Acoustic Edge Drone Detection (Raspberry Pi 4)
Documentation=https://github.com/<org>/sky-fort-acoustic
After=network.target sound.target
Wants=sound.target

[Service]
Type=simple
User=skyfort-edge
Group=skyfort-edge
SupplementaryGroups=audio gpio

# Working dir and exec
WorkingDirectory=/opt/skyfort-edge
ExecStart=/opt/skyfort-edge/.venv/bin/python -m skyfort_edge --config /etc/skyfort-edge/config.yaml
Environment=PYTHONUNBUFFERED=1
Environment=PYTHONPATH=/opt/skyfort-edge

# Restart policy
Restart=on-failure
RestartSec=5s
StartLimitInterval=60s
StartLimitBurst=3

# Security hardening (T-21-07)
NoNewPrivileges=true
ProtectSystem=strict
ProtectHome=true
PrivateTmp=true
ProtectKernelTunables=true
ProtectKernelModules=true
ProtectControlGroups=true
RestrictSUIDSGID=true
RestrictNamespaces=true
LockPersonality=true
MemoryDenyWriteExecute=false
# MemoryDenyWriteExecute must stay false — onnxruntime uses JIT in some paths

# Writable paths (only these)
ReadWritePaths=/var/lib/skyfort-edge
ReadWritePaths=/var/log/skyfort-edge

# Resource limits
LimitNOFILE=4096
TasksMax=64

# Logging
StandardOutput=journal
StandardError=journal
SyslogIdentifier=skyfort-edge

[Install]
WantedBy=multi-user.target
```

Create `scripts/install_edge_rpi.sh` (bash, idempotent, run on Pi with sudo):

```bash
#!/usr/bin/env bash
# Sky Fort Edge install script for Raspberry Pi 4 (Bookworm, Python 3.11).
# Idempotent: safe to re-run. Installs venv, deps, systemd unit, default config.
#
# Usage:  sudo bash scripts/install_edge_rpi.sh [REPO_ROOT]
# If REPO_ROOT is not given, defaults to $(pwd) assuming the repo is checked out locally.

set -euo pipefail

REPO_ROOT="${1:-$(pwd)}"
APP_DIR="/opt/skyfort-edge"
CONFIG_DIR="/etc/skyfort-edge"
DATA_DIR="/var/lib/skyfort-edge"
LOG_DIR="/var/log/skyfort-edge"
SERVICE_USER="skyfort-edge"
UNIT_SRC="${REPO_ROOT}/apps/rpi-edge/systemd/skyfort-edge.service"
UNIT_DST="/etc/systemd/system/skyfort-edge.service"

log() { echo "[install_edge_rpi] $*"; }
fail() { echo "[install_edge_rpi] ERROR: $*" >&2; exit 1; }

[[ $EUID -eq 0 ]] || fail "must run as root (use sudo)"
[[ -d "${REPO_ROOT}/apps/rpi-edge" ]] || fail "REPO_ROOT=${REPO_ROOT} does not contain apps/rpi-edge/"
[[ -f "${UNIT_SRC}" ]] || fail "missing systemd unit at ${UNIT_SRC}"

log "step 1/8: apt packages"
apt-get update -q
apt-get install -y --no-install-recommends \
    python3.11 \
    python3.11-venv \
    libportaudio2 \
    libasound2 \
    lgpio \
    ca-certificates

log "step 2/8: system user"
if ! id -u "${SERVICE_USER}" >/dev/null 2>&1; then
    useradd --system --home-dir "${APP_DIR}" --shell /usr/sbin/nologin --groups audio,gpio "${SERVICE_USER}"
fi

log "step 3/8: directories"
install -d -o "${SERVICE_USER}" -g "${SERVICE_USER}" -m 0755 "${APP_DIR}"
install -d -o "${SERVICE_USER}" -g "${SERVICE_USER}" -m 0755 "${APP_DIR}/models"
install -d -o "${SERVICE_USER}" -g "${SERVICE_USER}" -m 0750 "${DATA_DIR}"
install -d -o "${SERVICE_USER}" -g "${SERVICE_USER}" -m 0750 "${LOG_DIR}"
install -d -o root -g root -m 0755 "${CONFIG_DIR}"

log "step 4/8: copy edge app sources (rsync-style idempotent)"
# Copy apps/rpi-edge/skyfort_edge/ package + assets
install -d -m 0755 "${APP_DIR}/skyfort_edge"
cp -r "${REPO_ROOT}/apps/rpi-edge/skyfort_edge/." "${APP_DIR}/skyfort_edge/"
install -d -m 0755 "${APP_DIR}/assets"
cp "${REPO_ROOT}/apps/rpi-edge/assets/alert.wav" "${APP_DIR}/assets/alert.wav"
install -m 0644 "${REPO_ROOT}/apps/rpi-edge/requirements.txt" "${APP_DIR}/requirements.txt"

log "step 5/8: copy ONNX model artifacts + checksums"
install -m 0644 "${REPO_ROOT}/models/efficientat_mn10_v6_fp32.onnx" "${APP_DIR}/models/efficientat_mn10_v6_fp32.onnx"
install -m 0644 "${REPO_ROOT}/models/efficientat_mn10_v6_int8.onnx" "${APP_DIR}/models/efficientat_mn10_v6_int8.onnx"
install -m 0644 "${REPO_ROOT}/models/efficientat_mn10_v6_onnx.sha256" "${APP_DIR}/models/efficientat_mn10_v6_onnx.sha256"
(cd "${APP_DIR}/models" && sha256sum -c efficientat_mn10_v6_onnx.sha256) || fail "ONNX checksum verification failed"

log "step 6/8: default config at ${CONFIG_DIR}/config.yaml (only if missing)"
if [[ ! -f "${CONFIG_DIR}/config.yaml" ]]; then
    install -m 0644 "${REPO_ROOT}/apps/rpi-edge/config.yaml" "${CONFIG_DIR}/config.yaml"
    log "  installed default config"
else
    log "  existing config preserved"
fi

log "step 7/8: python venv + pinned deps"
if [[ ! -d "${APP_DIR}/.venv" ]]; then
    sudo -u "${SERVICE_USER}" python3.11 -m venv "${APP_DIR}/.venv"
fi
sudo -u "${SERVICE_USER}" "${APP_DIR}/.venv/bin/pip" install --upgrade pip
sudo -u "${SERVICE_USER}" "${APP_DIR}/.venv/bin/pip" install -r "${APP_DIR}/requirements.txt"

# Smoke-check: import onnxruntime
sudo -u "${SERVICE_USER}" "${APP_DIR}/.venv/bin/python" -c \
    "import onnxruntime, sys; print('onnxruntime', onnxruntime.__version__)" \
    || fail "onnxruntime import failed on Pi — check wheel architecture / CPU support"

chown -R "${SERVICE_USER}:${SERVICE_USER}" "${APP_DIR}"

log "step 8/8: systemd unit"
install -m 0644 "${UNIT_SRC}" "${UNIT_DST}"
systemctl daemon-reload
systemctl enable skyfort-edge.service
log "install complete. Start with: sudo systemctl start skyfort-edge"
log "monitor with:            journalctl -u skyfort-edge -f"
log "view detections with:    sudo tail -f ${DATA_DIR}/detections.jsonl"
```

# Sky Fort Edge (Raspberry Pi 4)

Standalone edge drone-detection app running the EfficientAT MN10 v6 model via ONNX Runtime on a bare Raspberry Pi 4. Captures audio from a single USB microphone, runs continuous classification, drives a GPIO LED on detection, and writes a rotating JSONL detection log.

## Features

- Single USB mic, 48 kHz → 32 kHz resample → 128-mel → ONNX int8 inference
- Hysteresis state machine with configurable enter/exit thresholds + confirm/release hit counts
- GPIO LED alarm with `min_on_seconds` latch (SIGTERM-safe cleanup)
- Optional audio alarm (bundled alert.wav, disabled by default, silent-degrade)
- Always-on rotating JSONL detection log (cannot be disabled via config)
- localhost-only `/health` + `/status` HTTP endpoint (stdlib, no FastAPI)
- Runs as a systemd service on boot

## Hardware Requirements

- Raspberry Pi 4 (tested on 4GB, Bookworm 64-bit)
- USB class-compliant microphone
- Single LED + current-limiting resistor on configurable GPIO pin (default: GPIO 17)
- Optional: speaker/headphones on default ALSA sink

## Install (on the Pi)

SSH into the Pi as a user with sudo privileges, then:

```bash
git clone https://github.com/<org>/sky-fort-acoustic.git
cd sky-fort-acoustic
sudo bash scripts/install_edge_rpi.sh
sudo systemctl start skyfort-edge
```

The install script is idempotent — safe to re-run after updating the repo.

## Monitoring

General application logs go to journald (D-23):
```bash
journalctl -u skyfort-edge -f
```

Detection events go to the rotating JSONL log (cannot be disabled):
```bash
sudo tail -f /var/lib/skyfort-edge/detections.jsonl
```

Live status endpoint (localhost only):
```bash
curl http://127.0.0.1:8088/status
curl http://127.0.0.1:8088/health
```

## Config Tweaks

Edit `/etc/skyfort-edge/config.yaml` and restart the service. No hot-reload (D-10):
```bash
sudo nano /etc/skyfort-edge/config.yaml
sudo systemctl restart skyfort-edge
```

Common tweaks:
- `thresholds.enter_threshold` — raise to reduce false positives, lower to catch weaker signals
- `hardware.led_gpio_pin` — change the GPIO BCM pin number
- `hardware.min_on_seconds` — how long the LED stays on after the last positive frame
- `hardware.alarm_enabled: true` — enable the bundled audio alarm
- `model.prefer_int8: false` — force FP32 fallback (larger, slower, higher accuracy)

## Troubleshooting

**Service won't start:** `journalctl -u skyfort-edge -n 100` — look for config validation errors or ONNX load failures.

**No USB mic found:** `arecord -l` to list capture devices. Set `hardware.input_device` in config.yaml to the device name.

**"illegal instruction" on startup:** onnxruntime 1.18.1 is pinned for Pi 4 Cortex-A72 compatibility. If this still fails, the wheel may be corrupt — re-run `sudo bash scripts/install_edge_rpi.sh`.

**LED stuck on after crash:** SIGTERM handlers release GPIO cleanly, but `kill -9` bypasses them. If the pin is stuck, reboot or manually toggle with `gpioset`.

**Detection log growing unbounded:** Rotation is configured in `detection_log.rotate_max_bytes` and `rotate_backup_count`. Default: 10 MB × 5 = 50 MB max.

## Architecture Note

The edge app lives in `apps/rpi-edge/` and intentionally does NOT import from the main `src/acoustic/` service package (D-25). Preprocessing math is vendored as a pure-numpy reimplementation (`skyfort_edge/preprocess.py`) with a CI drift test (`tests/integration/test_edge_preprocess_drift.py`) that fails loudly if the training-side reference diverges.
