"""Shared mutable state between main loop and HTTP handler. Thread-safe."""
from __future__ import annotations
import threading
from dataclasses import dataclass, field, asdict
from typing import Optional


@dataclass
class RuntimeState:
    model_loaded: bool = False
    audio_stream_alive: bool = False
    last_inference_time: Optional[float] = None
    last_detection_time: Optional[float] = None
    led_state: str = "off"
    log_file_path: str = ""
    active_model_path: str = ""

    def __post_init__(self):
        self._lock = threading.Lock()

    def snapshot(self) -> dict:
        with self._lock:
            d = asdict(self)
        d.pop("_lock", None)
        return d

    def update(self, **kwargs) -> None:
        with self._lock:
            for k, v in kwargs.items():
                if hasattr(self, k) and not k.startswith("_"):
                    setattr(self, k, v)

"""D-24 + T-21-01: Minimal localhost-only JSON /health + /status endpoint."""
from __future__ import annotations
import json
import logging
import threading
from http.server import BaseHTTPRequestHandler, HTTPServer
from typing import Optional

from skyfort_edge.runtime_state import RuntimeState

log = logging.getLogger(__name__)

LOOPBACK_HOSTS = {"127.0.0.1", "localhost", "::1"}


class _Handler(BaseHTTPRequestHandler):
    state: Optional[RuntimeState] = None  # set by factory

    def log_message(self, format, *args):
        # Route to Python logger, not stderr
        log.debug("%s - %s", self.client_address[0], format % args)

    def _send_json(self, status_code: int, body: dict) -> None:
        payload = json.dumps(body).encode("utf-8")
        self.send_response(status_code)
        self.send_header("Content-Type", "application/json")
        self.send_header("Content-Length", str(len(payload)))
        self.end_headers()
        self.wfile.write(payload)

    def do_GET(self):  # noqa: N802
        if self.path == "/health":
            snap = self.state.snapshot() if self.state else {}
            status = "ok" if snap.get("model_loaded") and snap.get("audio_stream_alive") else "degraded"
            self._send_json(200, {
                "status": status,
                "model_loaded": snap.get("model_loaded", False),
                "audio_stream_alive": snap.get("audio_stream_alive", False),
            })
        elif self.path == "/status":
            snap = self.state.snapshot() if self.state else {}
            self._send_json(200, snap)
        else:
            self._send_json(404, {"error": "not found", "path": self.path})


class LocalhostJSONServer:
    def __init__(self, http_cfg, state: RuntimeState) -> None:
        if http_cfg.bind_host not in LOOPBACK_HOSTS:
            raise ValueError(
                f"T-21-01: bind_host must be loopback (got {http_cfg.bind_host!r})"
            )
        _Handler.state = state
        self._server = HTTPServer((http_cfg.bind_host, http_cfg.bind_port), _Handler)
        self._thread: Optional[threading.Thread] = None

    def start(self) -> None:
        self._thread = threading.Thread(target=self._server.serve_forever, daemon=True)
        self._thread.start()
        log.info("HTTP server listening on %s", self._server.server_address)

    def stop(self) -> None:
        self._server.shutdown()
        self._server.server_close()
        if self._thread is not None:
            self._thread.join(timeout=2.0)

"""Sky Fort Edge main entrypoint.

Usage:
    python -m skyfort_edge --config /etc/skyfort-edge/config.yaml [--log-level INFO]

Architecture (composition root):
    AudioCapture (48k) → resample_poly(2,3) → NumpyMelSTFT → OnnxClassifier →
        score → HysteresisStateMachine → LedAlarm + AudioAlarm + DetectionLogger
    LocalhostJSONServer exposes /health + /status in parallel.
"""
from __future__ import annotations
import argparse
import logging
import signal
import sys
import time
from pathlib import Path
from typing import Optional

import numpy as np

from skyfort_edge.config import load_config, EdgeConfig
from skyfort_edge.audio import AudioCapture
from skyfort_edge.preprocess import NumpyMelSTFT
from skyfort_edge.inference import OnnxClassifier
from skyfort_edge.hysteresis import HysteresisStateMachine, EventType
from skyfort_edge.gpio_led import LedAlarm
from skyfort_edge.audio_alarm import AudioAlarm
from skyfort_edge.detection_log import DetectionLogger
from skyfort_edge.http_server import LocalhostJSONServer
from skyfort_edge.runtime_state import RuntimeState

log = logging.getLogger("skyfort_edge")
ALERT_WAV_PATH = Path(__file__).parent.parent / "assets" / "alert.wav"


def _parse_args(argv=None) -> argparse.Namespace:
    p = argparse.ArgumentParser(prog="skyfort_edge")
    p.add_argument("--config", type=Path, default=None, help="Path to YAML config")
    p.add_argument("--log-level", default=None, help="Override model.log_level")
    p.add_argument("--score-threshold", type=float, default=None)
    p.add_argument("--enter-threshold", type=float, default=None)
    p.add_argument("--exit-threshold", type=float, default=None)
    p.add_argument("--led-gpio-pin", type=int, default=None)
    p.add_argument("--dry-run", action="store_true", help="Start everything then exit after 1 iteration")
    return p.parse_args(argv)


def _cli_to_overrides(args: argparse.Namespace) -> dict:
    o: dict = {}
    if args.score_threshold is not None:
        o.setdefault("thresholds", {})["score_threshold"] = args.score_threshold
    if args.enter_threshold is not None:
        o.setdefault("thresholds", {})["enter_threshold"] = args.enter_threshold
    if args.exit_threshold is not None:
        o.setdefault("thresholds", {})["exit_threshold"] = args.exit_threshold
    if args.led_gpio_pin is not None:
        o.setdefault("hardware", {})["led_gpio_pin"] = args.led_gpio_pin
    if args.log_level is not None:
        o.setdefault("model", {})["log_level"] = args.log_level
    return o


def _setup_general_logging(level: str) -> None:
    logging.basicConfig(
        level=getattr(logging, level.upper(), logging.INFO),
        format="%(asctime)s %(levelname)s %(name)s: %(message)s",
    )


def _score_from_logits(logits: np.ndarray, num_classes: Optional[int]) -> tuple[str, float]:
    """Return (class_name, score). For binary: score = sigmoid(logit[0]). For multi-class: softmax argmax."""
    if num_classes == 1 or logits.shape[0] == 1:
        sig = 1.0 / (1.0 + float(np.exp(-logits[0])))
        return ("drone", float(sig))
    exp = np.exp(logits - np.max(logits))
    probs = exp / exp.sum()
    idx = int(np.argmax(probs))
    return (f"class_{idx}", float(probs[idx]))


class EdgeApp:
    def __init__(self, cfg: EdgeConfig) -> None:
        self.cfg = cfg
        self.state = RuntimeState(log_file_path=cfg.detection_log.path)
        self.preprocess = NumpyMelSTFT()
        self.classifier = OnnxClassifier(cfg.model)
        self.state.update(model_loaded=True, active_model_path=str(self.classifier.active_model_path))
        self.hysteresis = HysteresisStateMachine(
            enter_threshold=cfg.thresholds.enter_threshold,
            exit_threshold=cfg.thresholds.exit_threshold,
            confirm_hits=cfg.thresholds.confirm_hits,
            release_hits=cfg.thresholds.release_hits,
            min_on_seconds=cfg.hardware.min_on_seconds,
        )
        self.led = LedAlarm(gpio_pin=cfg.hardware.led_gpio_pin)
        self.alarm = AudioAlarm(
            enabled=cfg.hardware.alarm_enabled,
            alert_wav_path=ALERT_WAV_PATH,
            device=cfg.hardware.alarm_audio_device,
        )
        self.det_log = DetectionLogger(cfg.detection_log)
        self.http = LocalhostJSONServer(cfg.http, self.state)
        self.audio: Optional[AudioCapture] = None
        self._stop = False

    def _process_window(self, wave_32k: np.ndarray) -> None:
        mel = self.preprocess.forward(wave_32k)
        logits = self.classifier.classify(mel)
        class_name, score = _score_from_logits(logits, self.classifier.num_classes)
        now = time.time()
        self.state.update(last_inference_time=now)
        event = self.hysteresis.update(score, now)
        if event is None:
            return
        if event.type == EventType.RISING_EDGE:
            self.led.on_event(event)
            self.alarm.play()
            self.det_log.write_latch(event, class_name=class_name, score=score)
            self.state.update(last_detection_time=now, led_state="on")
        elif event.type == EventType.FALLING_EDGE:
            self.led.on_event(event)
            self.alarm.reset()
            self.det_log.write_latch(event, class_name=class_name, score=score)
            self.state.update(led_state="off")

    def run(self) -> None:
        self.audio = AudioCapture(device=self.cfg.hardware.input_device)
        self.audio.start()
        self.state.update(audio_stream_alive=True)
        self.http.start()

        signal.signal(signal.SIGTERM, lambda *a: setattr(self, "_stop", True))
        signal.signal(signal.SIGINT, lambda *a: setattr(self, "_stop", True))

        hop = self.cfg.timing.hop_seconds
        window = self.cfg.timing.window_seconds
        next_tick = time.time()
        log.info("EdgeApp running: window=%.2fs hop=%.2fs", window, hop)
        while not self._stop:
            try:
                wave = self.audio.read_window_32k(window)
                self._process_window(wave)
            except Exception as e:
                log.exception("process_window error: %s", e)
            next_tick += hop
            sleep = next_tick - time.time()
            if sleep > 0:
                time.sleep(sleep)
            else:
                next_tick = time.time()
        self.shutdown()

    def shutdown(self) -> None:
        log.info("EdgeApp shutting down")
        try:
            if self.audio is not None:
                self.audio.stop()
        except Exception:
            pass
        try:
            self.http.stop()
        except Exception:
            pass
        try:
            self.led.close()
        except Exception:
            pass
        try:
            self.det_log.close()
        except Exception:
            pass


def main(argv=None) -> int:
    args = _parse_args(argv)
    try:
        cfg = load_config(yaml_path=args.config, cli_overrides=_cli_to_overrides(args))
    except Exception as e:
        print(f"Config error: {e}", file=sys.stderr)
        return 2
    _setup_general_logging(cfg.model.log_level)
    app = EdgeApp(cfg)
    if args.dry_run:
        app.shutdown()
        return 0
    try:
        app.run()
    except KeyboardInterrupt:
        app.shutdown()
    return 0


if __name__ == "__main__":
    raise SystemExit(main())
